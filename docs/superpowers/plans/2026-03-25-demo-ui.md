```python
"""Shared test helpers for woven-imprint tests."""

from woven_imprint.engine import Engine


class FakeLLM:
    def __init__(self):
        self.call_count = 0

    def generate(self, messages, **kw):
        self.call_count += 1
        return "I hear you."

    def generate_json(self, messages, **kw):
        self.call_count += 1
        system = messages[0].get("content", "") if messages else ""
        if "extract" in system.lower() or "fact" in system.lower():
            return ["A notable fact was shared"]
        if "emotion" in system.lower():
            return {"mood": "neutral", "intensity": 0.5, "cause": ""}
        if "relationship" in system.lower():
            return {"trust": 0.01, "affection": 0.0, "respect": 0.0,
                    "familiarity": 0.02, "tension": 0.0}
        if "narrative" in system.lower() or "beat" in system.lower():
            return {"beat_type": "none"}
        if "consistency" in system.lower():
            return {"score": 1.0, "issues": []}
        if "summar" in system.lower():
            return "Session summary"
        return {}


class FakeEmbedder:
    def __init__(self):
        self._vocab = {}
        self._next = 0

    def embed(self, text):
        vec = [0.0] * 50
        for word in text.lower().split()[:10]:
            if word not in self._vocab:
                self._vocab[word] = self._next % 50
                self._next += 1
            vec[self._vocab[word]] += 1.0
        mag = sum(x * x for x in vec) ** 0.5
        if mag > 0:
            vec = [x / mag for x in vec]
        return vec

    def embed_batch(self, texts):
        return [self.embed(t) for t in texts]

    def dimensions(self):
        return 50


def make_test_engine(db_path=":memory:"):
    """Create an Engine with fakes, parallel disabled."""
    llm = FakeLLM()
    embedder = FakeEmbedder()
    engine = Engine(db_path=db_path, llm=llm, embedding=embedder)
    orig = engine.create_character
    def _create_seq(*a, **kw):
        c = orig(*a, **kw)
        c.parallel = False
        return c
    engine.create_character = _create_seq
    return engine
```

```python
"""Service-layer functions for woven-imprint server endpoints.

Business logic extracted from sidecar.py and api.py handlers.
Both the existing stdlib servers and the new FastAPI demo server
call these same functions — no logic duplication.
"""

from __future__ import annotations

from typing import Any

from woven_imprint.engine import Engine


# ---------------------------------------------------------------------------
# Character services
# ---------------------------------------------------------------------------

def create_character_service(
    engine: Engine,
    name: str,
    persona: dict | str | None,
    birthdate: str | None,
) -> dict[str, Any]:
    """Create a character, deduplicating by name (case-insensitive).

    Returns dict with keys: id, name, created (bool).
    """
    # Check for existing character with same name
    existing = engine.list_characters()
    for char_info in existing:
        if char_info["name"].lower() == name.lower():
            return {"id": char_info["id"], "name": char_info["name"], "created": False}

    # Normalize persona: string → dict with personality key
    # (matches existing sidecar.py behavior for backwards compat)
    if isinstance(persona, str):
        persona = {"personality": persona}

    char = engine.create_character(name=name, persona=persona, birthdate=birthdate)
    return {"id": char.id, "name": name, "created": True}


def list_characters_service(engine: Engine) -> list[dict[str, Any]]:
    """List all characters."""
    return engine.list_characters()


def get_character_state_service(
    engine: Engine,
    character_id: str,
) -> dict[str, Any]:
    """Get character info including emotion and arc state.

    Raises KeyError if character not found.
    """
    char = engine.get_character(character_id)
    emotion = char.emotion.to_dict() if hasattr(char, "emotion") and char.emotion else {}
    arc_phase = ""
    arc_tension = 0.0
    if hasattr(char, "arc") and char.arc:
        arc_phase = char.arc.current_phase.value if hasattr(char.arc.current_phase, "value") else str(char.arc.current_phase)
        arc_tension = getattr(char.arc, "tension", 0.0)

    # Use char.name property (delegates to persona.name), NOT char.persona.persona
    return {
        "id": char.id,
        "name": getattr(char, "name", character_id),
        "emotion": emotion,
        "arc": {"phase": arc_phase, "tension": arc_tension},
    }


# ---------------------------------------------------------------------------
# Session services
# ---------------------------------------------------------------------------

def start_session_service(engine: Engine, character_id: str) -> dict[str, str]:
    """Start a new session for a character. Returns {session_id: str}."""
    char = engine.get_character(character_id)
    session_id = char.start_session()
    return {"session_id": session_id}


def end_session_service(engine: Engine, character_id: str) -> dict[str, Any]:
    """End the current session. Returns {summary: str | None}."""
    char = engine.get_character(character_id)
    summary = char.end_session()
    return {"summary": summary}


# ---------------------------------------------------------------------------
# Message services
# ---------------------------------------------------------------------------

_SIDECAR_ROLES = {"user", "assistant"}
_ALL_ROLES = {"user", "assistant", "character", "system"}


def record_message_service(
    engine: Engine,
    character_id: str,
    role: str,
    content: str,
    user_id: str | None,
    *,
    strict_roles: bool = True,
) -> None:
    """Record a message via ingest (no LLM generation).

    Args:
        strict_roles: If True (default, matches sidecar behavior), only
            accept "user" and "assistant". Set False for demo server
            which also accepts "character" and "system".

    Raises ValueError for invalid role.
    """
    allowed = _SIDECAR_ROLES if strict_roles else _ALL_ROLES
    if role not in allowed:
        raise ValueError(f"Invalid role '{role}'. Must be one of: {allowed}")
    char = engine.get_character(character_id)
    char.ingest(role=role, content=content, user_id=user_id)


def recall_memories_service(
    engine: Engine,
    character_id: str,
    query: str,
    limit: int = 10,
    user_id: str | None = None,
) -> dict[str, Any]:
    """Recall memories and build context string.

    Returns {memories: list[dict], context: str}.
    """
    char = engine.get_character(character_id)
    memories = char.recall(query, limit=limit)

    # Build context string (extracted from sidecar.py lines 257-277)
    context_parts: list[str] = []
    if user_id:
        rel = char.get_relationship(user_id)
        if rel:
            desc = char.relationships.describe(user_id)
            context_parts.append(f"[Relationship with {user_id}: {desc}]")
    if memories:
        context_parts.append("[Relevant memories:]")
        for m in memories[:limit]:
            content = m.get("content", "")[:200]
            context_parts.append(f"- {content}")
    context = "\n".join(context_parts)

    return {"memories": memories, "context": context}


# ---------------------------------------------------------------------------
# Relationship services
# ---------------------------------------------------------------------------

def get_relationship_service(
    engine: Engine,
    character_id: str,
    target_id: str,
) -> dict[str, Any] | None:
    """Get relationship dimensions between character and target.

    Returns relationship dict or None if no relationship exists.
    """
    char = engine.get_character(character_id)
    return char.get_relationship(target_id)


# ---------------------------------------------------------------------------
# Chat services (for OpenAI-compatible endpoint)
# ---------------------------------------------------------------------------

def find_character_by_name_or_id(
    engine: Engine,
    name_or_id: str,
) -> dict[str, Any] | None:
    """Find a character by name variants or exact ID.

    Tries: lowercase match, dash-replaced, underscore-replaced, exact ID.
    """
    chars = engine.list_characters()
    model_name = name_or_id.lower().strip()
    return next(
        (c for c in chars
         if c["name"].lower().replace(" ", "-") == model_name
         or c["name"].lower().replace(" ", "_") == model_name
         or c["name"].lower() == model_name
         or c["id"] == name_or_id),
        None,
    )


def extract_last_user_message(messages: list[dict]) -> str:
    """Extract the last user message from an OpenAI-format message list."""
    for msg in reversed(messages):
        if msg.get("role") == "user":
            return msg.get("content", "")
    return ""


def extract_user_id_from_messages(
    messages: list[dict],
    default: str = "api_user",
) -> str:
    """Extract user_id from system messages (format: 'user_id: <id>')."""
    for msg in messages:
        if msg.get("role") == "system" and "user_id:" in msg.get("content", ""):
            for line in msg["content"].split("\n"):
                if line.strip().startswith("user_id:"):
                    return line.split(":", 1)[1].strip()
    return default
```

```python
"""FastAPI demo server for woven-imprint.

Single-process server that serves:
- /api/* — sidecar logic (character state, memory, relationships)
- /v1/* — OpenAI-compatible chat completions
- /api/config/* — provider configuration

from woven_imprint.config import get_config

_auth_token: str = ""
_mutation_lock: asyncio.Lock = asyncio.Lock()

STATIC_DIR = Path(__file__).parent.parent / "demo_static"


def _get_engine() -> Engine:
    if _engine is None:
        raise HTTPException(503, "Engine not initialized")
    return _engine


def _check_auth(request: Request) -> None:
    auth = request.headers.get("authorization", "")
    if not auth.startswith("Bearer ") or auth[7:] != _auth_token:
        raise HTTPException(401, "Invalid or missing auth token")


def create_app(
    engine: Engine | None = None,
    port: int = 7860,
    token: str | None = None,
) -> tuple[FastAPI, str]:
    """Create the FastAPI demo app.

    Returns (app, auth_token).
    """
    global _engine, _auth_token

    _auth_token = token or secrets.token_urlsafe(32)
    if engine is not None:
        _engine = engine

    app = FastAPI(title="woven-imprint demo", version=__version__)

    # CORS — localhost only
    app.add_middleware(
        CORSMiddleware,
        allow_origins=[f"http://127.0.0.1:{port}", f"http://localhost:{port}"],
        allow_methods=["GET", "POST", "DELETE", "OPTIONS"],
        allow_headers=["Authorization", "Content-Type"],
    )

    # --- Health (no auth) ---

    @app.get("/api/health")
    async def health():
        return {"status": "ok", "version": __version__}

    # --- Character endpoints (auth required) ---

    @app.get("/api/characters")
    async def list_characters(_: None = Depends(_check_auth)):
        eng = _get_engine()
        return {"characters": list_characters_service(eng)}

    @app.post("/api/characters")
    async def create_character(
        body: CreateCharacterRequest,
        _: None = Depends(_check_auth),
    ):
        eng = _get_engine()
        async with _mutation_lock:
            result = create_character_service(eng, body.name, body.persona, body.birthdate)
        status = 201 if result["created"] else 200
        return JSONResponse(result, status_code=status)

    @app.get("/api/characters/{character_id}")
    async def get_character(character_id: str, _: None = Depends(_check_auth)):
        eng = _get_engine()
        try:
            return get_character_state_service(eng, character_id)
        except KeyError:
            raise HTTPException(404, f"Character '{character_id}' not found")

    # --- Session endpoints ---

    @app.post("/api/characters/{character_id}/session")
    async def start_session(character_id: str, _: None = Depends(_check_auth)):
        eng = _get_engine()
        async with _mutation_lock:
            try:
                return start_session_service(eng, character_id)
            except KeyError:
                raise HTTPException(404, f"Character '{character_id}' not found")

    @app.delete("/api/characters/{character_id}/session")
    async def end_session(character_id: str, _: None = Depends(_check_auth)):
        eng = _get_engine()
        async with _mutation_lock:
            try:
                return end_session_service(eng, character_id)
            except KeyError:
                raise HTTPException(404, f"Character '{character_id}' not found")

    # --- Record ---

    @app.post("/api/record")
    async def record_message(
        body: RecordMessageRequest,
        _: None = Depends(_check_auth),
    ):
        eng = _get_engine()
        async with _mutation_lock:
            try:
                record_message_service(eng, body.character_id, body.role, body.content, body.user_id)
            except KeyError:
                raise HTTPException(404, f"Character '{body.character_id}' not found")
            except ValueError as exc:
                raise HTTPException(400, str(exc))
        return {"ok": True}

    # --- Memory ---

    @app.get("/api/memory")
    async def recall_memories(
        character_id: str,
        query: str,
        limit: int = 10,
        user_id: str | None = None,
        _: None = Depends(_check_auth),
    ):
        eng = _get_engine()
        try:
            return recall_memories_service(eng, character_id, query, limit, user_id)
        except KeyError:
            raise HTTPException(404, f"Character '{character_id}' not found")

    # --- Relationships ---

    @app.get("/api/relationships/{character_id}/{target_id}")
    async def get_relationship(
        character_id: str,
        target_id: str,
        _: None = Depends(_check_auth),
    ):
        eng = _get_engine()
        try:
            rel = get_relationship_service(eng, character_id, target_id)
            return {"relationship": rel}
        except KeyError:
            raise HTTPException(404, f"Character '{character_id}' not found")

    # --- Chat completions (OpenAI-compatible) ---

    @app.post("/v1/chat/completions")
    async def chat_completions(
        body: ChatCompletionRequest,
        _: None = Depends(_check_auth),
    ):
        eng = _get_engine()
        match = find_character_by_name_or_id(eng, body.model)
        if not match:
            raise HTTPException(404, f"Character '{body.model}' not found")

        user_msg = extract_last_user_message(body.messages)
        if not user_msg:
            raise HTTPException(400, "No user message found")

        user_id = extract_user_id_from_messages(body.messages)
        char = eng.get_character(match["id"])

        async with _mutation_lock:
            response = char.chat(user_msg, user_id=user_id)

        emotion = char.emotion.to_dict() if hasattr(char, "emotion") and char.emotion else {}
        arc_phase = ""
        arc_tension = 0.0
        if hasattr(char, "arc") and char.arc:
            arc_phase = char.arc.current_phase.value if hasattr(char.arc.current_phase, "value") else str(char.arc.current_phase)
            arc_tension = getattr(char.arc, "tension", 0.0)

        return {
            "id": f"chatcmpl-{uuid.uuid4().hex[:12]}",
            "object": "chat.completion",
            "created": int(time.time()),
            "model": body.model,
            "choices": [{
                "index": 0,
                "message": {"role": "assistant", "content": response},
                "finish_reason": "stop",
            }],
            "usage": {
                "prompt_tokens": len(user_msg) // 4,
                "completion_tokens": len(response) // 4,
                "total_tokens": (len(user_msg) + len(response)) // 4,
            },
            "woven_imprint": {
                "character": match["name"],
                "emotion": emotion,
                "arc": {"phase": arc_phase, "tension": arc_tension},
            },
        }

    # --- Provider config ---

    @app.get("/api/config/provider")
    async def get_provider_config(_: None = Depends(_check_auth)):
        cfg = get_config()
        return {
            "provider": cfg.llm.llm_provider or "ollama",
            "model": cfg.llm.model or "",
            "base_url": cfg.llm.base_url,
            "api_key_configured": bool(cfg.llm.api_key),
        }

    @app.post("/api/config/provider")
    async def update_provider_config(
        body: ProviderConfigRequest,
        _: None = Depends(_check_auth),
    ):
        # Write to config file
        import yaml
        config_path = Path.home() / ".woven_imprint" / "config.yaml"
        config_path.parent.mkdir(parents=True, exist_ok=True)

        existing = {}
        if config_path.exists():
            existing = yaml.safe_load(config_path.read_text()) or {}

        existing.setdefault("llm", {})
        existing["llm"]["llm_provider"] = body.provider
        existing["llm"]["model"] = body.model
        if body.base_url:
            existing["llm"]["base_url"] = body.base_url
        if body.api_key:
            existing["llm"]["api_key"] = body.api_key

        config_path.write_text(yaml.dump(existing, default_flow_style=False))

        # Reload config AND rebuild the engine's LLM so changes take
        # effect without a server restart (W3 fix)
        from woven_imprint.config import reload_config
        reload_config()
        if _engine:
            new_cfg = get_config()
            from woven_imprint.providers import create_llm
            _engine._llm = create_llm(cfg=new_cfg)

        return {
            "provider": body.provider,
            "model": body.model,
            "base_url": body.base_url,
            "api_key_configured": bool(body.api_key),
        }

    @app.post("/api/config/provider/test")
    async def test_provider_connection(
        body: ProviderConfigRequest,
        _: None = Depends(_check_auth),
    ):
        try:
            # Build a transient WovenConfig to pass to create_llm

            test_cfg.llm = LLMConfig(
                llm_provider=body.provider,

            )

            # Simple test: generate a short response
            llm.generate([{"role": "user", "content": "Say hello"}], max_tokens=10)
            return {"success": True, "message": "Connection successful"}
        except Exception as exc:
            # Never expose raw stack trace or secrets
            msg = str(exc)
            # Strip anything that looks like an API key
            for prefix in ("sk-", "nvapi-", "key-"):
                if prefix in msg:
                    msg = "Connection failed. Check your API key and provider settings."
                    break
            return {"success": False, "message": msg}

    # --- Token injection for frontend (W6 fix) ---
    # Serve index.html with injected auth token instead of static file

    @app.get("/")
    async def index_with_token():
        index_path = STATIC_DIR / "index.html"
        if not index_path.exists():
            return JSONResponse({"error": "Demo UI not built. Run: cd demo && npm run build"}, 503)
        html = index_path.read_text()
        # Inject token as global variable before closing </head>
        inject = f'<script>window.__WOVEN_TOKEN__="{_auth_token}";</script>'
        html = html.replace("</head>", f"{inject}</head>", 1)
        from starlette.responses import HTMLResponse
        return HTMLResponse(html)

    # --- Static files (catch-all, excludes root which is handled above) ---

    if STATIC_DIR.exists():
        app.mount("/", StaticFiles(directory=str(STATIC_DIR), html=False), name="static")

    return app, _auth_token


def run_demo_server(
    port: int = 7860,
    db_path: str | None = None,
    model: str | None = None,
    no_browser: bool = False,
) -> None:
    """Start the demo server. Called by CLI."""
    import uvicorn
    import webbrowser

    global _engine

    cfg = get_config()
    resolved_db = db_path or cfg.storage.db_path
    resolved_model = model or cfg.llm.model

    from woven_imprint.providers import create_llm, create_embedding
    llm = create_llm(cfg=cfg)
    embedder = create_embedding(cfg=cfg)
    _engine = Engine(db_path=resolved_db, llm=llm, embedding=embedder)

    # Seed Meridian if not present
    _seed_meridian_if_needed(_engine)

    app, token = create_app(engine=_engine, port=port)

    print(f"\n  woven-imprint demo server")
    print(f"  URL:   http://127.0.0.1:{port}")
    print(f"  Token: {token}")
    print(f"  Press Ctrl+C to stop\n")

    if not no_browser:
        webbrowser.open(f"http://127.0.0.1:{port}")

    try:
        uvicorn.run(app, host="127.0.0.1", port=port, log_level="warning")
    finally:
        if _engine:
            _engine.close()


def _seed_meridian_if_needed(engine: Engine) -> None:

    engine.create_character(
        name="Meridian",
        persona=MERIDIAN_PERSONA,
        birthdate=MERIDIAN_BIRTHDATE,
    )
```

```python
class FakeLLM:
    def generate(self, messages, **kw):
        return "I hear you."

    def generate_json(self, messages, **kw):
        system = messages[0].get("content", "") if messages else ""
        if "extract" in system.lower() or "fact" in system.lower():
            return ["A notable fact"]
        if "emotion" in system.lower():
            return {"mood": "neutral", "intensity": 0.5, "cause": ""}
        if "relationship" in system.lower():
            return {"trust": 0.01, "affection": 0.0, "respect": 0.0,
                    "familiarity": 0.02, "tension": 0.0}
        if "narrative" in system.lower() or "beat" in system.lower():
            return {"beat_type": "none"}
        if "consistency" in system.lower():
            return {"score": 1.0, "issues": []}
        if "summar" in system.lower():
            return "Session summary"
        return {}


class FakeEmbedder:
    def __init__(self):
        self._vocab = {}
        self._next = 0

    def embed(self, text):
        vec = [0.0] * 50
        for word in text.lower().split()[:10]:
            if word not in self._vocab:
                self._vocab[word] = self._next % 50
                self._next += 1
            vec[self._vocab[word]] += 1.0
        mag = sum(x * x for x in vec) ** 0.5
        if mag > 0:
            vec = [x / mag for x in vec]
        return vec

    def embed_batch(self, texts):
        return [self.embed(t) for t in texts]

    def dimensions(self):
        return 50


def _make_engine(db_path):
    llm = FakeLLM()
    embedder = FakeEmbedder()
    engine = Engine(db_path=db_path, llm=llm, embedding=embedder)
    orig = engine.create_character
    def _create_seq(*a, **kw):
        c = orig(*a, **kw)
        c.parallel = False
        return c
    engine.create_character = _create_seq
    return engine


class TestCrossSessionPersistence:
    """Verify that facts survive across sessions and engine restarts."""

    def test_memory_survives_session_restart(self):
        """Fact from session 1 is recalled in session 2."""
        with tempfile.NamedTemporaryFile(suffix=".db", delete=False) as f:
            db_path = f.name
        try:
            # Session 1: create character and record a fact
            engine1 = _make_engine(db_path)
            r = create_character_service(engine1, "TestChar", None, None)
            char_id = r["id"]
            start_session_service(engine1, char_id)
            record_message_service(engine1, char_id, "user", "My favorite color is blue", None)
            end_session_service(engine1, char_id)
            engine1.close()

            # Session 2: new engine instance, recall the fact
            engine2 = _make_engine(db_path)
            result = recall_memories_service(engine2, char_id, "favorite color", 10, None)
            contents = [m.get("content", "") for m in result["memories"]]
            assert any("blue" in c.lower() for c in contents), \
                f"Expected 'blue' in recalled memories, got: {contents}"
            engine2.close()
        finally:
            os.unlink(db_path)

    def test_character_survives_restart(self):
        """Character created in session 1 exists in session 2."""
        with tempfile.NamedTemporaryFile(suffix=".db", delete=False) as f:
            db_path = f.name
        try:
            engine1 = _make_engine(db_path)
            r = create_character_service(engine1, "Persistent", None, None)
            char_id = r["id"]
            engine1.close()

            engine2 = _make_engine(db_path)
            chars = engine2.list_characters()
            assert any(c["id"] == char_id for c in chars)
            engine2.close()
        finally:
            os.unlink(db_path)
```

```python
def _import_seed_db(engine: Engine, seed_path: Path) -> None:
    """Import Meridian character data from pre-built seed database.

    Copies character record and all memories from seed DB into the
    user's database. The seed DB was generated by build_meridian_seed.py.
    """
    import sqlite3
    import json

    seed_conn = sqlite3.connect(str(seed_path))
    seed_conn.row_factory = sqlite3.Row

    # Get Meridian's character record from seed
    row = seed_conn.execute(
        "SELECT * FROM characters WHERE name = 'Meridian' LIMIT 1"
    ).fetchone()
    if not row:
        seed_conn.close()
        return

    char_id = row["id"]
    persona = json.loads(row["persona"]) if row["persona"] else {}

    # Create character in user's DB (uses engine's LLM for embeddings)
    from woven_imprint.data.meridian_persona import MERIDIAN_PERSONA, MERIDIAN_BIRTHDATE
    char = engine.create_character(
        name="Meridian",
        persona=MERIDIAN_PERSONA,
        birthdate=MERIDIAN_BIRTHDATE,
        character_id=char_id,
    )

    # Import pre-chunked bedrock memories from seed
    memories = seed_conn.execute(
        "SELECT content, tier, importance FROM memories WHERE character_id = ?",
        (char_id,),
    ).fetchall()

    for mem in memories:
        char.memory.add(
            content=mem["content"],
            tier=mem["tier"],
            importance=mem["importance"],
        )

    seed_conn.close()


def _seed_meridian_if_needed(engine: Engine) -> None:
    chars = engine.list_characters()
    for c in chars:
        if c["name"].lower() == "meridian":
            return

    # Try seed DB first
    seed_db = Path(__file__).parent.parent / "data" / "meridian_seed.db"
    if seed_db.exists():
        _import_seed_db(engine, seed_db)
        return

    # Fallback: create with persona only (no pre-built knowledge)
    from woven_imprint.data.meridian_persona import MERIDIAN_PERSONA, MERIDIAN_BIRTHDATE
    engine.create_character(
        name="Meridian",
        persona=MERIDIAN_PERSONA,
        birthdate=MERIDIAN_BIRTHDATE,
```
